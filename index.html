# A small function to count the number of a set of chars in a string

def countChars( message, fset="aeiou" ):
  count = 0
  for el in message:
    if el in fset:
      count = count + 1
  return count

print( countChars("Hello World", "o" ) )
print( countChars("I study at NKUST" ) )

def checkIncludes( lst, fset = [ 1, 2, 3 ] ):
  res = True
  for f in fset:
    fnd = False
    for l in lst:
      if ( l == f ):
        fnd = True
        break  
    if not fnd:
      res = False
      break
  return res

import random

def randomIntList( N, min, max ):
  lst = []
  for i in range(N):
    lst.append( random.randint(min, max) )
  return lst

lst = randomIntList( 5, -9, 10 )

def sumEven( lst ):
  sum = 0
  for l in lst:
    if l % 2 == 0:

Sum Even After Odd Numbers

Implement a function that returns the sum of all even numbers that follow immediately after an odd number in a list:

sumEvenAfterOdd( [ 1, 2, 4, 7, -2, 3, -11 ] ) => 0

def sumEvenAfterOdd( lst ):
  ...

def sumEvenAfterOdd( lst ):
  sum = 0
  for i in range(1, len(lst)):
    if lst[i] % 2 == 0 and lst[i-1] % 2 != 0:
      sum = sum + lst[i]
  return sum

for k in range(3):
  peval( "sumEvenAfterOdd(" + str( randomIntList(5, -9, 10 ) ) + ")" )
jb """Since we need to look at our neighbors in the list. It is better to use the index i (for i in range(len(lst) - 1)), since we can then peek at the next number easily. """ 
def sumEvenAfterOdd( lst ):
  sum = 0
  prev = 0
  for el in lst:
    if el % 2 == 0 and prev % 2 != 0:
      sum = sum + el
    prev = el
  return sum

for k in range(3):
  peval( "sumEvenAfterOdd(" + str( randomIntList(5, -9, 10 ) ) + ")" )
jb "This version of the program uses a for-each loop, but remembers" jb """the previous element. It is slightly more efficient and elegant than the first version"""
